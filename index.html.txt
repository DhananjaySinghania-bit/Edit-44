// Splash & reveal
(function(){
  const splash=document.getElementById('splash');
  const header=document.getElementById('siteHeader');
  const main=document.getElementById('main');
  const sections=main.querySelectorAll('section');
  const buttons=document.querySelector('.buttons');
  const footer=document.querySelector('footer');

  window.addEventListener('load',()=>{
    setTimeout(()=>{
      splash.style.opacity='0';
      setTimeout(()=>splash.remove(),500);

      header.classList.add('visible'); header.removeAttribute('aria-hidden');
      sections.forEach((s,i)=> setTimeout(()=>s.classList.add('visible'), i*150));
      setTimeout(()=> buttons.classList.add('visible'), sections.length*150+150);
      setTimeout(()=> footer.classList.add('visible'), sections.length*150+300);
      main.removeAttribute('aria-hidden');
    },1500);
  });

  window.addEventListener('scroll',()=>{
    const trigger = window.innerHeight*0.85;
    sections.forEach(s=>{
      if(!s.classList.contains('visible') && s.getBoundingClientRect().top<trigger) s.classList.add('visible');
    });
  });
})();

// Particle background
const canvas=document.getElementById('particles');
const ctx=canvas.getContext('2d');
let W=canvas.width=window.innerWidth;
let H=canvas.height=window.innerHeight;
let particles=[];
for(let i=0;i<100;i++){
  particles.push({x:Math.random()*W, y:Math.random()*H, r:Math.random()*2+1, dx:(Math.random()-0.5)*0.7, dy:(Math.random()-0.5)*0.7});
}
function draw(){
  ctx.clearRect(0,0,W,H);
  particles.forEach(p=>{
    ctx.beginPath();
    ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
    ctx.fillStyle="rgba(0,255,255,0.6)";
    ctx.fill();
    p.x+=p.dx; p.y+=p.dy;
    if(p.x<0||p.x>W)p.dx*=-1;
    if(p.y<0||p.y>H)p.dy*=-1;
  });
  requestAnimationFrame(draw);
}
draw();
window.addEventListener('resize',()=>{W=canvas.width=window.innerWidth;H=canvas.height=window.innerHeight;});
